```typescript
import {
  createDefaultPromptRegistry,
  getActivePrompt,
  forkPrompt,
  compareRuns,
} from './prompt-registry'

describe('prompt registry', () => {
  it('seeds default prompts for all 10 phases', () => {
    const registry = createDefaultPromptRegistry()
    for (let phase = 1; phase <= 10; phase++) {
      expect(registry.active_versions[phase]).toBeDefined()
      expect(registry.official_versions[phase]).toBeDefined()
      const prompt = registry.versions[registry.active_versions[phase]]
      expect(prompt).toBeDefined()
      expect(prompt.phase).toBe(phase)
      expect(prompt.is_official).toBe(true)
      expect(prompt.content.length).toBeGreaterThan(0)
    }
  })

  it('returns the active prompt for a phase', () => {
    const registry = createDefaultPromptRegistry()
    const prompt = getActivePrompt(registry, 6)
    expect(prompt.phase).toBe(6)
    expect(prompt.is_official).toBe(true)
  })

  it('falls back to a default prompt for an unregistered phase', () => {
    const registry = createDefaultPromptRegistry()
    const prompt = getActivePrompt(registry, 99)
    expect(prompt.phase).toBe(99)
    expect(prompt.is_official).toBe(true)
  })

  it('forks a prompt and sets it as active', () => {
    const registry = createDefaultPromptRegistry()
    const { registry: updated, version } = forkPrompt(registry, 3, {
      name: 'Custom Phase 3',
      content: 'Custom instructions for baseline modeling.',
      description: 'Testing fork',
    })
    expect(version.phase).toBe(3)
    expect(version.is_official).toBe(false)
    expect(version.name).toBe('Custom Phase 3')
    expect(version.content).toBe('Custom instructions for baseline modeling.')
    expect(version.parent_id).toBe(registry.active_versions[3])
    expect(updated.active_versions[3]).toBe(version.id)
    // Official version unchanged
    expect(updated.official_versions[3]).toBe(registry.official_versions[3])
  })

  it('forks with defaults when no params provided', () => {
    const registry = createDefaultPromptRegistry()
    const original = getActivePrompt(registry, 1)
    const { version } = forkPrompt(registry, 1)
    expect(version.name).toBe(`${original.name} fork`)
    expect(version.content).toBe(original.content)
  })

  it('compareRuns detects prompt version changes', () => {
    const registry = createDefaultPromptRegistry()
    const { registry: updated, version: forked } = forkPrompt(registry, 2)

    const leftExecution = {
      id: 'exec_1', phase: 2, pass_number: 1,
      provider_id: 'test', model_id: 'test',
      prompt_version_id: registry.active_versions[2],
      started_at: '', completed_at: null, duration_ms: null,
      input_tokens: 0, output_tokens: 0, cost_usd: null,
      status: 'complete' as const, error: null,
    }
    const rightExecution = { ...leftExecution, id: 'exec_2', prompt_version_id: forked.id }

    const comparison = compareRuns(updated, leftExecution, rightExecution)
    expect(comparison.changed).toBe(true)
    expect(comparison.phase).toBe(2)
  })

  it('compareRuns reports no change for same version', () => {
    const registry = createDefaultPromptRegistry()
    const execution = {
      id: 'exec_1', phase: 5, pass_number: 1,
      provider_id: 'test', model_id: 'test',
      prompt_version_id: registry.active_versions[5],
      started_at: '', completed_at: null, duration_ms: null,
      input_tokens: 0, output_tokens: 0, cost_usd: null,
      status: 'complete' as const, error: null,
    }
    const comparison = compareRuns(registry, execution, execution)
    expect(comparison.changed).toBe(false)
  })
})
```

## Chunk 2: Fix revalidation trigger phase mappings

The revalidation engine uses fixed `affected_phases` per source phase, ignoring which specific trigger was found. The spec defines trigger-specific target phases.

### Current vs expected mappings

| Trigger | Current affected_phases | Spec target | Fix |
|---------|------------------------|-------------|-----|
| `new_player_discovered` (from P2) | [3, 4] | [2] | Keep [3, 4] — Phase 2 just completed; downstream needs rerun |
| `objective_function_changed` (from P4) | [3, 4] | [2, 3] | Change to [2, 3] — Phase 2 must update the player's objectives |
| `new_game_identified` (from P3) | [3, 4] | [3] | Keep [3, 4] — Phase 4 also needs the new game |
| `game_reframed` (from P6) | [3, 4, 6] | [3] | Narrow to [3, 6] — Phase 4 may not need full rerun for reframing |
| `new_cross_game_link` (from P6) | [6] | [3i or 6i] | Keep [6] — cross-game effects are Phase 6i scope |
| `behavioral_overlay_changes_prediction` (from P6) | [6] | [6h] | Keep [6] — correct granularity for phase reruns |
| `critical_empirical_assumption_invalidated` (from P7) | [7, 8, 9] | [7] | Keep [7, 8, 9] — spec says "loops to Phase 7, then re-runs Phase 8 + 9" |

Only `objective_function_changed` needs a definite fix. The `game_reframed` change from [3,4,6] to [3,6] is a refinement.

Also: `getPhase6Check` uses fixed affected_phases for all triggers from Phase 6. It should be trigger-specific.

```typescript
function getPhase4Check(result: HistoricalGameResult): RevalidationCheck {
  if (!result.baseline_recheck.revalidation_needed) {
    return createCheck([], [], [], 'none', 'Phase 4 historical review converged with the current baseline.')
  }

  const gameRefs = result.dynamic_inconsistency_risks.flatMap((risk) => risk.affected_games)
  const playerRefs = result.trust_assessment.flatMap((assessment) => [
    createEntityRef('player', assessment.assessor_player_id),

  const triggers = result.baseline_recheck.revalidation_triggers

    }
  }

  return createCheck(
    triggers,
    affectedPhases,
    [...gameRefs, ...playerRefs],
```

- [ ] **Step 1: Add "Implementation Deviations" section to 6.1 spec**

```markdown
## Implementation Deviations

1. **Prompt registry storage**: Spec says SQLite. Implementation uses Zustand + browser
   persistence adapter (localStorage). Rationale: no SQLite infrastructure exists in the
   app; localStorage provides equivalent persistence for a local-first desktop app.
   The adapter can be swapped for SQLite later without API changes.

2. **RevalidationEngine.checkTriggers signature**: Spec takes `PhaseResult`;
   implementation takes `unknown` (the full phase output object, not just PhaseResult).
   Rationale: trigger detection needs the full phase output (e.g., proposed_players,
   baseline_recheck), not just the status summary.

3. **Trigger #1 (new_player_discovered) routes to [3,4] not [2]**: Phase 2 just
   completed when this fires, so re-running Phase 2 is redundant. Downstream phases
   (3, 4) need to incorporate the new player.
```

- [ ] **Step 2: Add "Implementation Deviations" section to 6.2 spec**

```markdown
## Implementation Deviations

1. **Flat result structure vs nested sub_results**: Spec uses
   `FormalizationResult.sub_results.representations`. Implementation uses flat fields
   (`formal_representations`, `payoff_estimation`, etc.) plus `subsection_statuses`,
   `proposal_groups`, and `workspace_previews`. Rationale: flat fields are simpler to
   destructure and the additional fields (subsection_statuses, workspace_previews) don't
   fit the spec's `sub_results` nesting.

2. **Sub-result types are summary-oriented**: Spec contracts define per-entity detail
   types (e.g., `ProposedFormalization` with full structure). Implementation uses summary
   types (e.g., `FormalizationRepresentationSummary`) because full detail is in the
   canonical store after proposal acceptance. Phase results are status snapshots, not
   the canonical data itself.

3. **PayoffEstimationResult**: Spec has `estimates: PayoffEstimate[]` with
   `ordinal_rank: string`. Implementation has `updates: PayoffEstimationSummary[]`
   with `ordinal_first: boolean`. Ordinal rankings are stored on the formalization
   entities themselves (canonical layer), not duplicated in the phase result.
```

- [ ] **Step 3: Add "Implementation Deviations" section to 6.3 spec**

```markdown
## Implementation Deviations

1. **SensitivitySummary renamed to Phase7SensitivitySummary**: Avoids name collision
   with potential M4 solver sensitivity types. Fields are identical to spec.

2. **Owner attachment system**: Implementation adds `OwnerAttachment` types
   (game, formalization, game_node, game_edge, normal_form_estimate, etc.) not in
   the spec. Rationale: assumptions need fine-grained ownership tracking to specific
   payoff cells, nodes, and edges — not just game-level references.
```
